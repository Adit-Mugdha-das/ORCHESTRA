symphony Math {
  flow add take(a, b) {
    return a + b;
  }

  flow mul take(a, b) {
    return a * b;
  }
}

ensemble Person {
  int age;
  string name;
}

symphony Text {
  flow hello take() {
    return "hi";
  }
}

symphony Box {
  int w;
  int h;

  flow area take(b) {
    return b.w * b.h;
  }
}

flow main take() {
  emit Math.add(2, 3);
  emit Math.mul(6, 7);
  emit Text.hello();

  note p = Person(20, "Ali");
  emit p.age;
  emit p.name;
  emit p;

  note b = Box(6, 7);
  emit b;
  emit Box.area(b);
}
