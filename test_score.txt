// Test 1: basic score (for) loop — count 1 to 5
flow main {
    score (note i = 1; i <= 5; stage i = i + 1) {
        emit i;
    }

    // Test 2: nested score loops — multiplication table 2x2
    emit "--- nested ---";
    score (note row = 1; row <= 3; stage row = row + 1) {
        score (note col = 1; col <= 3; stage col = col + 1) {
            emit row * col;
        }
    }

    // Test 3: continue in score loop — skip value 3 and 7
    emit "--- skip 3 and 7 ---";
    score (note n = 1; n <= 9; stage n = n + 1) {
        branch (n == 3 || n == 7) {
            continue;
        }
        emit n;
    }

    // Test 4: break inside score loop
    emit "--- break at 4 ---";
    score (note k = 1; k <= 10; stage k = k + 1) {
        branch (k == 4) {
            break;
        }
        emit k;
    }

    // Test 5: score with no init (step only)
    emit "--- no init ---";
    note x = 10;
    score (; x > 0; stage x = x - 3) {
        emit x;
    }
}
