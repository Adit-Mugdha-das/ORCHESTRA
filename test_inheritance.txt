symphony Shape {
  int x;
  int y;

  flow init take(x, y) {
    stage this.x = x;
    stage this.y = y;
    return 0;
  }

  flow move take(dx, dy) {
    stage this.x = this.x + dx;
    stage this.y = this.y + dy;
    return 0;
  }

  flow area take() {
    return this.x + this.y;
  }
}

symphony Rect extends Shape {
  int w;
  int h;

  flow init take(x, y, w, h) {
    super(x, y);
    stage this.w = w;
    stage this.h = h;
    return 0;
  }

  flow area take() {
    return super.area() + (this.w * this.h);
  }
}

flow main take() {
  note r = Rect(1, 2, 6, 7);
  emit r;

  r.move(3, 4);
  emit r;

  emit r.area();
}
